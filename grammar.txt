#variable list
L -> I | I,L

# declarations
D -> int L

#read
R -> read I

#write
W -> write I | write N

#expression
-----------------------------
E -> E + T1 | E - T1 | T1
T1 -> T1 * T2 | T1 / T2 | T2
T2 -> T2 > T3 | T2 == T3 | T3
T3 -> (E) | I | N
------------------------------

#assignment
A -> I = E

#for loop
F -> for(A; E; A){S}

#program
P -> D;S | S

#statements
S -> A; | R; | W; | F; | A;S | R;S | W;S | F;S
